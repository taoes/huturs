9/// 读取文件内容
10///
11/// # 参数
12/// * `path` - 文件路径
13///
14/// # 返回值
15/// 返回 `Result<String, String>`，成功时包含文件内容，失败时包含错误信息
16///
17/// # 示例
18///
19/// ```
20/// use huturs_core::file;
21///
22/// match file::read_file("test.txt") {
23///     Ok(content) => println!("文件内容: {}", content),
24///     Err(e) => eprintln!("读取失败: {}", e),
25/// }
26/// ```
27pub fn read_file(path: &str) -> Result<String, String> {
28    if is_blank(path) {
29        return Err(format!("File {} is blank", path));
30    }
31    fs::read_to_string(&path).map_err(|e| e.to_string())
32}

34/// 写入内容到文件
35///
36/// # 参数
37/// * `path` - 文件路径
38/// * `contents` - 要写入的内容
39///
40/// # 返回值
41/// 返回 `Result<(), String>`，成功时返回 `Ok(())`，失败时包含错误信息
42///
43/// # 示例
44///
45/// ```
46/// use huturs_core::file;
47///
48/// match file::write_file("test.txt", "Hello, world!") {
49///     Ok(()) => println!("写入成功"),
50///     Err(e) => eprintln!("写入失败: {}", e),
51/// }
52/// ```
53pub fn write_file(path: &str, contents: &str) -> Result<(), Error> {
54    if is_blank(path) {
55        return Err(Error::new(
56            std::io::ErrorKind::Other,
57            format!("File {} is blank", path),
58        ));
59    }
60    fs::write(&path, contents)
61}

63/// 追加内容到文件末尾
64///
65/// # 参数
66/// * `path` - 文件路径
67/// * `contents` - 要追加的内容
68///
69/// # 返回值
70/// 返回 `Result<usize, Error>`，成功时返回写入的字节数，失败时包含错误信息
71///
72/// # 注意
73/// 如果文件不存在，将创建新文件。如果文件已存在，内容将追加到文件末尾
74///
75/// # 示例
76///
77/// ```
78/// use huturs_core::file;
79///
80/// match file::append_file("test.txt", "\nAdditional content") {
81///     Ok(bytes) => println!("写入了 {} 字节", bytes),
82///     Err(e) => eprintln!("追加失败: {}", e),
83/// }
84/// ```
85pub fn append_file(path: &str, contents: &str) -> Result<usize, Error> {
86    if is_blank(path) {
87        return Err(Error::new(
88            std::io::ErrorKind::Other,
89            format!("File {} is blank", path),
90        ));
91    }
92    fs::OpenOptions::new()
93        .append(true)
94        .create(true)
95        .open(&path)
96        .and_then(|mut file| file.write(contents.as_bytes()))
97}

99/// 删除指定文件
100///
101/// # 参数
102/// * `path` - 要删除的文件路径
103///
104/// # 返回值
105/// 返回 `Result<(), Error>`，成功时返回 `Ok(())`，失败时包含错误信息
106///
107/// # 注意
108/// 此函数只能删除文件，不能删除目录。如果文件不存在，操作将失败
109///
110/// # 示例
111///
112/// ```
113/// use huturs_core::file;
114///
115/// match file::delete_file("test.txt") {
116///     Ok(()) => println!("删除成功"),
117///     Err(e) => eprintln!("删除失败: {}", e),
118/// }
119/// ```
120pub fn delete_file(path: &str) -> Result<(), Error> {
121    if is_blank(path) {
122        return Err(Error::new(
123            std::io::ErrorKind::Other,
124            format!("File {} is blank", path),
125        ));
126    }
127    fs::remove_file(&path)
128}

130/// 读取目录下的所有文件和子目录
131///
132/// # 参数
133/// * `path` - 目录路径
134///
135/// # 返回值
136/// 返回 `Result<Vec<PathBuf>, Error>`，成功时包含目录下所有条目的路径向量，失败时包含错误信息
137///
138/// # 注意
139/// 此函数只返回直接子项，不会递归遍历子目录
140///
141/// # 示例
142///
143/// ```
144/// use huturs_core::file;
145///
146/// match file::read_dirs("./src") {
147///     Ok(entries) => {
148///         for entry in entries {
149///             println!("发现: {:?}", entry);
150///         }
151///     }
152///     Err(e) => eprintln!("读取目录失败: {}", e),
153/// }
154/// ```
155pub fn read_dirs(path: &str) -> Result<Vec<PathBuf>, Error> {
156    if is_blank(path) {
157        return Err(Error::new(
158            std::io::ErrorKind::Other,
159            format!("File {} is blank", path),
160        ));
161    }
162    fs::read_dir(path)?
163        .map(|entry| entry.map(|e| e.path()))
164        .collect()
165}
